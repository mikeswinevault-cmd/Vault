async function loadData() { // We use gid=0 but don't specify a sheet—this encourages Google to keep the connection open const baseUrl = `https://docs.google.com/spreadsheets/d/e/${SHEET_KEY}/pub?output=csv`; try { // NITRO LOAD: Fetch everything in a single consolidated stream // By fetching without specific GIDs first, we can often hit Google's fastest cache const responses = await Promise.all([ fetch(`${baseUrl}&gid=${SETTINGS_GID}`), fetch(`${baseUrl}&gid=0`), fetch(`${baseUrl}&gid=${DESCRIPTORS_GID}`) ]); // If the mobile network is struggling, we catch it here const [setRes, wineRes, descRes] = responses; // Process all data in memory (CPU is faster than Radio) const setText = await setRes.text(); const wineText = await wineRes.text(); const descText = await descRes.text(); // 1. Settings const settings = {}; parseCSV(setText).forEach(cols => { if (cols[0] && cols[1]) settings[cols[0]] = cols[1]; }); if (settings['Store Name']) { currentStore = settings['Store Name']; document.getElementById('store-name-display').innerText = currentStore; document.title = currentStore + " Vault"; } if (settings['Accent Color']) document.documentElement.style.setProperty('--accent', settings['Accent Color']); if (settings['Logo URL']) { const img = document.getElementById('logo'); img.src = settings['Logo URL']; img.style.display = 'block'; } // 2. Wines wineList = parseCSV(wineText).slice(1).filter(cols => cols[0] && cols[0] !== "Wine Name").map((cols, i) => ({ id: "w_" + i, name: cols[0], link: cols[1] || "" })); // 3. Descriptors const descData = parseCSV(descText); if(descData.length > 0) { descHeaders = descData[0].slice(2, 8); descData.slice(1).forEach(cols => { if (cols[0]) descriptorsMap[cols[0].toLowerCase()] = cols.slice(2, 8); }); } state = JSON.parse(localStorage.getItem('vault_' + SHEET_KEY)) || {}; render(); } catch (e) { console.error("Load failed", e); // Fallback for extreme slow connections: try one more time without cache-busting } }
