async function loadData() { try { // 1. UNIVERSAL WINE LOAD (Always the first tab) const wineUrl = `https://docs.google.com/spreadsheets/d/e/${SHEET_KEY}/pub?output=csv&gid=0&cb=${Date.now()}`; const wineRes = await fetch(wineUrl); const wineText = await wineRes.text(); const wineData = parseCSV(wineText); wineList = wineData.slice(1).filter(cols => cols[0] && cols[0] !== "Wine Name").map((cols, i) => ({ id: "w_" + i, name: cols[0], link: cols[1] || "" })); // 2. SMART SETTINGS LOAD (Finds the tab named "Settings" automatically) const settingsUrl = `https://docs.google.com/spreadsheets/d/e/${SHEET_KEY}/pub?output=csv&sheet=Settings&cb=${Date.now()}`; const settingsRes = await fetch(settingsUrl); const settingsText = await settingsRes.text(); const settingsData = parseCSV(settingsText); const settings = {}; settingsData.forEach(cols => { if (cols[0] && cols[1]) settings[cols[0].trim()] = cols[1].trim(); }); // Apply the "Napa" or "Ansonia" Branding if (settings['Store Name']) { storeName = settings['Store Name']; document.getElementById('store-name-display').innerText = storeName; document.getElementById('page-title').innerText = storeName + " Vault"; } if (settings['Accent Color']) { document.documentElement.style.setProperty('--accent', settings['Accent Color']); } if (settings['Logo URL']) { const img = document.getElementById('logo'); img.src = settings['Logo URL']; img.style.display = 'block'; } state = JSON.parse(localStorage.getItem('vault_' + SHEET_KEY)) || {}; document.getElementById('timestamp').innerText = "Sync: " + new Date().toLocaleTimeString(); render(); } catch (e) { console.error("Data Load Error", e); } }
